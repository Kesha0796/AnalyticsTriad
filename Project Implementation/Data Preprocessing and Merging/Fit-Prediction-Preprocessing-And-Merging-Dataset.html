def get_height_in_cms(height):
    if type(height) == type(1.0):
        return
    try: 
        h_inch = int(height[4:-2])
        h_inch += int(height[0]) * 12
        h_cm = round(h_inch * 2.54, 1)        
        return h_cm
    except:
        return (int(height[0]) * 12 * 2.54)

def get_height_cms(height):

        return

        h_inch += int(height[0]) * 12
